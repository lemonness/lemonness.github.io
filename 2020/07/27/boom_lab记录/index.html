3.根据索引查找链表

401176:	48 8b 52 08          	mov    0x8(%rdx),%rdx

上面为最关键的一行,也是确定数据结构的一行,根据这一行我们推测这题是一个链表结构,每一个链表节点有一个成员变量是一个指向下一个链表节点的指针,我们便能根据地址打印相关数据

例如: x/24xw 0x6032d0 打印链表的结构 表示打印24个单元,以十六进制,单元大小为4字节

在进行链表节点地址重排后,可以查看%rsp+0x20处后面的数组结构,发现是按照我们传入的数据进行节点地址的排序.

最后一步为比较大小,按照value值降序排列,否则触发bomb,综上分析,我们根据汇编写一下类似的C源码

/* structure node */
struct node{
int value; // 某四字节内容
int order; // 原来节点的顺序 
struct node* next; // 下一个node地址
} ;

/*C code in phtase 6*/
void phase_7(......){
	//part1
	int i;
	for(i=0;i<6;i++){
		int tmp = intarray[i];
		if(tmp>6) callbomb();  //每个数必须小于6
		
		for(;i<6;i++){ 
			if(tmp=intarray[i]) callbomb(); 每个数都不相同
		}
	}
	
	//part2
	for(i=0;i<6;i++){
		intarray[i]=7-intarray[i];
	}
	
	//part3 找到节点地址，并排序
	struct node node*;
	node* nodearray[6];
	int tmpb;
	for(i=0;i<6;i++){
		tmp = intarray[i];
		for(tmpb=0;tmpb<tmp;tmpb++){
			node=node->next;
		}
		nodearray[i]=node;
	}
	
	//part4 重新排列node的下一个索引地址，注意最后一个节点的下一个索引为0
	for(i=0;i<5;i++){
		node = nodearray[i];
		node->next = nodearray[i+1];
	}
	nodearray[i]=0;  //最后一个节点置零
	
	//part5 比较大小，只需要比较5次
	for(i=0;i<5;i++){
		node = nodearray[i];
		if(node.value<node->next.value) call_bomb();
	}
}

第7题:

本lab还有一个彩蛋关卡.

在defuse的两处设置断点,可以看出在0x4015d8出,会对0x6和0x202181(%rsp)进行对比,这里我们发现每解开一个程序后者会加1,所以解完前6个炸弹后,才会进入下面的判断代码

进入判断后,发现是用第四题的输入做判断 我们修改第四题的输入 即进入了彩蛋关卡

进入后,我们打印在0x6030f0处的值 发现是一个二叉树结构

二叉树结构

根据fun7的汇编,转换成C语言的代码

//node pointer in %rdi,value in %rsi,return_value in %rax 
//require %rax to be 2
int fun7(Node* node,int value)
{
  if(value == 0){
    return -1;
  }
  int v = addr->value;
  if( v < value){
        return 1 + 2*fun7(addr->right,value);
  }else if (v == value){
    return 0;
  }else {
    return 2*func7(addr->left,value);
  }
}

根据递归关系 ,我们要获得2的结果,则关系式为2*(1+2*0),则输入值为22或20
